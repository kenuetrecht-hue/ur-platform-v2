import React, { useState } from 'react';
import { View, Text, TextInput, Pressable, ScrollView, Image } from 'react-native';
import { ScreenContainer } from '@/components/screen-container';
import { useColors } from '@/hooks/use-colors';

interface LoginFormData {
  email: string;
  password: string;
}

export default function LoginScreen() {
  const colors = useColors();
  const [formData, setFormData] = useState<LoginFormData>({
    email: '',
    password: '',
  });
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const handleLogin = async () => {
    setError(null);
    setIsLoading(true);

    try {
      // Validate inputs
      if (!formData.email || !formData.password) {
        setError('Please enter email and password');
        setIsLoading(false);
        return;
      }

      // TODO: Replace with actual API call
      console.log('Login attempt:', formData.email);

      // Simulate API call
      await new Promise((resolve) => setTimeout(resolve, 1000));

      // TODO: Store auth token and redirect to dashboard
      console.log('Login successful');
    } catch (err) {
      setError('Login failed. Please try again.');
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <ScreenContainer>
      <ScrollView contentContainerStyle={{ flexGrow: 1 }} showsVerticalScrollIndicator={false}>
        <View style={{ flex: 1, justifyContent: 'center', alignItems: 'center', paddingHorizontal: 20 }}>
          {/* Logo */}
          <View style={{ marginBottom: 40, alignItems: 'center' }}>
            <Text style={{ fontSize: 48, fontWeight: 'bold', color: colors.primary, marginBottom: 8 }}>
              UR
            </Text>
            <Text style={{ fontSize: 16, color: colors.muted, textAlign: 'center' }}>
              The Platform for Creators & Builders
            </Text>
          </View>

          {/* Error Message */}
          {error && (
            <View
              style={{
                width: '100%',
                backgroundColor: colors.error,
                padding: 12,
                borderRadius: 8,
                marginBottom: 20,
              }}
            >
              <Text style={{ color: '#fff', fontSize: 14 }}>{error}</Text>
            </View>
          )}

          {/* Email Input */}
          <TextInput
            style={{
              width: '100%',
              borderWidth: 1,
              borderColor: colors.border,
              borderRadius: 8,
              padding: 14,
              marginBottom: 16,
              backgroundColor: colors.background,
              color: colors.foreground,
              fontSize: 16,
            }}
            placeholder="Email"
            placeholderTextColor={colors.muted}
            value={formData.email}
            onChangeText={(text) => setFormData({ ...formData, email: text })}
            editable={!isLoading}
            keyboardType="email-address"
            autoCapitalize="none"
          />

          {/* Password Input */}
          <TextInput
            style={{
              width: '100%',
              borderWidth: 1,
              borderColor: colors.border,
              borderRadius: 8,
              padding: 14,
              marginBottom: 24,
              backgroundColor: colors.background,
              color: colors.foreground,
              fontSize: 16,
            }}
            placeholder="Password"
            placeholderTextColor={colors.muted}
            value={formData.password}
            onChangeText={(text) => setFormData({ ...formData, password: text })}
            secureTextEntry
            editable={!isLoading}
          />

          {/* Login Button */}
          <Pressable
            onPress={handleLogin}
            disabled={isLoading}
            style={({ pressed }) => [
              {
                width: '100%',
                backgroundColor: isLoading ? colors.muted : colors.primary,
                padding: 16,
                borderRadius: 8,
                alignItems: 'center',
                opacity: pressed ? 0.8 : 1,
              },
            ]}
          >
            <Text style={{ color: '#fff', fontSize: 16, fontWeight: 'bold' }}>
              {isLoading ? 'Logging in...' : 'Login'}
            </Text>
          </Pressable>

          {/* Sign Up Link */}
          <View style={{ marginTop: 20, flexDirection: 'row', justifyContent: 'center' }}>
            <Text style={{ color: colors.muted, fontSize: 14 }}>Don't have an account? </Text>
            <Pressable>
              <Text style={{ color: colors.primary, fontSize: 14, fontWeight: 'bold' }}>Sign Up</Text>
            </Pressable>
          </View>
        </View>
      </ScrollView>
    </ScreenContainer>
  );
}
